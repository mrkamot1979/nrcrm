CREATE TABLE tblclientdetails(
	ClientID INT NOT NULL AUTO_INCREMENT,
	fname TEXT NOT NULL,
	lname TEXT NOT NULL,
	address TEXT NOT NULL,
	gender TEXT NOT NULL,
	email TEXT NOT NULL,
	company TEXT NOT NULL,
	enteredby TEXT NOT NULL,
	PRIMARY KEY ( ClientID )
);

	interactionby text NOT NULL,
